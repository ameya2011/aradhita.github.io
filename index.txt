    <!-- DYNAMIC INTERACTION LOGIC SYSTEM -->
    <script>
        const card = document.getElementById('brochureCard');
        const canvas = document.getElementById('brochureCanvas');
        const immersiveBtn = document.getElementById('immersiveBtn');
        let isFlipped = false;
        let isImmersive = false;

        function toggleFlip() {
            isFlipped = !isFlipped;
            if (isFlipped) {
                card.classList.add('flipped');
            } else {
                card.classList.remove('flipped');
            }
        }

        function handleCardClick(e) {
            if (e.target.tagName.toLowerCase() === 'button' || e.target.closest('button')) return;
            toggleFlip();
        }

        function toggleImmersive() {
            isImmersive = !isImmersive;
            if (isImmersive) {
                canvas.style.position = 'fixed';
                canvas.style.inset = '0px';
                canvas.style.zIndex = '1000';
                canvas.style.backgroundColor = '#f6ebd9';
                canvas.style.backgroundImage = 'linear-gradient(rgba(231, 111, 81, 0.04) 1px, transparent 1px), linear-gradient(90deg, rgba(231, 111, 81, 0.04) 1px, transparent 1px)';
                canvas.style.backgroundSize = '20px 20px';
                canvas.style.width = '100vw';
                canvas.style.height = '100vh';
                canvas.style.maxWidth = '100%';
                canvas.style.padding = '60px';
                immersiveBtn.innerText = "🏽 Revert View";
                immersiveBtn.classList.add('bg-rose-900', 'text-white');
            } else {
                canvas.style.position = '';
                canvas.style.inset = '';
                canvas.style.zIndex = '';
                canvas.style.backgroundColor = '';
                canvas.style.backgroundImage = '';
                canvas.style.backgroundSize = '';
                canvas.style.width = '';
                canvas.style.height = '';
                canvas.style.maxWidth = '';
                canvas.style.padding = '';
                immersiveBtn.innerText = "✨ Maximize View";
                immersiveBtn.classList.remove('bg-rose-900', 'text-white');
            }
        }

        window.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && isImmersive) {
                toggleImmersive();
            }
        });
    </script>
